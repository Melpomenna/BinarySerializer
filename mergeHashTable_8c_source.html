164static hash_t DefaultMurmurHash2(const StatData *stData) {
165 hash_t m = 0xc6a4a7935bd1e995ULL;
166 const int r = 47;
167 hash_t h = 0x8445d61a4e774912ULL ^ (8 * m);
168
169 hash_t k = stData->id;
170 k *= m;
171 k ^= k >> r;
172 k *= m;
173
174 h ^= k;
175 h *= m;
176
177 h ^= h >> r;
178 h *= m;
179 h ^= h >> r;
180
181 return h;
182}

219static void DefaultMerge(StatData *__restrict first,
220 const StatData *__restrict second) {
221 // total sum of count
222 first->count += second->count;
223
224 // total sum of cost
225 first->cost += second->cost;
226
227 // if any primary have zero, than primary always zero
228 first->primary &= second->primary;
229
230 // max(mode) without if, for performance branch predictor
231 unsigned tmpFirstMode = first->mode;
232 unsigned tmpSecondMode = second->mode;
233 int ftrue = (tmpFirstMode > tmpSecondMode);
234 int ffalse = ftrue ^ 1;
235 first->mode = ftrue * tmpFirstMode + ffalse * tmpSecondMode;
236}

345BINARYSERIALIZER_NODISCARD static int insertIntoBucket(MergeHashTable *table,
346 Bucket *bucket,
347 const StatData *data,
348 hash_t hash) {
349 for (size_t i = 0; i < bucket->nodesCount; ++i) {
350 if (bucket->nodes[i].hash == hash &&
351 table->comparator(bucket->nodes[i].data, data) == 1) {
352 table->merge(bucket->nodes[i].data, data);
353 return 1;
354 }
355 }
356
357 Node *p = bucket->nodes;
358 StatData *newData = malloc(sizeof(StatData));
359 if (BINARYSERIALIZER_UNLIKELY(!newData)) {
360 return 0;
361 }
362 if (BINARYSERIALIZER_UNLIKELY(bucket->nodesCount == bucket->capacity ||
363 !bucket->nodes)) {
364 bucket->capacity *= 2;
365 p = realloc(bucket->nodes, sizeof(Node) * bucket->capacity);
366 if (BINARYSERIALIZER_UNLIKELY(!p)) {
367 free(newData);
368 return 0;
369 }
370 }
371
372 int nodesCount = bucket->nodesCount;
373 bucket->nodes = p;
374 p[nodesCount].hash = hash;
375 memcpy(newData, data, sizeof(StatData));
376 p[nodesCount].data = newData;
377 p[nodesCount].bucket = bucket;
378 p[nodesCount].index = nodesCount;
379 bucket->nodesCount++;
380 return 1;
381}

383int initHashTable(MergeHashTable *table, HashFunction hash, MergeFunction merge,
384 StatDataCompareFunction comparator) {
385 if (BINARYSERIALIZER_UNLIKELY(!table))
386 return 0;
387 table->hash = hash ? hash : &DefaultMurmurHash2;
388 table->merge = merge ? merge : &DefaultMerge;
389 table->comparator = comparator ? comparator : &DefaultStatDataComparator;
390 table->buckets =
391 malloc(sizeof(Bucket) * BINARYSERIALIZER_DEFUALT_BUCKETS_COUNT);
392 if (BINARYSERIALIZER_LIKELY(table->buckets)) {
393 table->bucketsCount = BINARYSERIALIZER_DEFUALT_BUCKETS_COUNT;
394 for (size_t i = 0; i < BINARYSERIALIZER_DEFUALT_BUCKETS_COUNT; ++i) {
395 table->buckets[i].nodes = NULL;
396 table->buckets[i].nodesCount = 0;
397 table->buckets[i].capacity = 1;
398 }
399 }
400 return table->buckets != NULL;
401}

413void eraseFromHashTable(MergeHashTable *table, const StatData *data) {
414 Node *node = findInHashTable(table, data);
415 if (BINARYSERIALIZER_UNLIKELY(!node))
416 return;
417
418 if (node->bucket->nodesCount - 1 == 0) {
419 clearBucket(node->bucket);
420 return;
421 }
422
423 size_t index = node->bucket->nodesCount - 1;
424 Node *lastNode = node->bucket->nodes + index;
425
426 if (node->hash != lastNode->hash) {
427 memcpy(node->data, lastNode->data, sizeof(StatData));
428 free(lastNode->data);
429 node->hash = lastNode->hash;
430 } else {
431 lastNode->index--;
432 }
433
434 node->bucket->nodesCount--;
435}

437Node *findInHashTable(const MergeHashTable *table, const StatData *data) {
438 if (BINARYSERIALIZER_UNLIKELY(!table || !data || !table->hash))
439 return NULL;
440
441 hash_t hash = table->hash(data);
442 size_t index = BucketIndex(table, hash);
443
444 Bucket *bucket = table->buckets + index;
445 for (size_t i = 0; i < bucket->nodesCount; ++i) {
446 if (bucket->nodes[i].hash == hash)
447 return bucket->nodes + i;
448 }
449
450 return NULL;
451}
